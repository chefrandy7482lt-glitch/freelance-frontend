// server.js const express = require('express'); const cors = require('cors'); const app = express(); const PORT = process.env.PORT || 10000; // Middleware app.use(cors()); app.use(express.json()); // In-memory job store (for demo) let jobs = []; // Health check app.get('/', (req, res) => { res.send('TDU Freelance Pipeline is running'); }); // Receive job packet from frontend app.post('/api/jobs', (req, res) => { const job = req.body; // Basic validation if (!job.task_group || !job.job_type || !job.details) { return res.status(400).json({ error: 'Missing required fields' }); } // Add server-side fields job.id = jobs.length + 1; job.status = 'received'; job.created_at = new Date().toISOString(); jobs.push(job); console.log('New job received:', job); // Simulate proposal + execution + result const responsePacket = { id: job.id, status: 'completed', proposal: { price: job.price || 75, currency: 'USD', }, result: { summary: `Job "${job.job_type}" processed successfully.`, echo_details: job.details, }, }; res.json(responsePacket); }); // Optional: list jobs (for dashboard later) app.get('/api/jobs', (req, res) => { res.json(jobs); }); app.listen(PORT, () => { console.log(`TDU Freelance Pipeline listening on port ${PORT}`); });
